```bash
#!/bin/bash
# =============================================================================
# rsync_to_a100.sh — 将本地 Evo-RL 代码同步到 A100
#
# 用法:
#   bash scripts/rsync_to_a100.sh
#
# 同步内容:
#   本地 Evo-RL/ → A100:/moganshan/afs_a/lai/Evo-RL/
#   排除: .git/ outputs/ pretrained/
#
# 注意:
#   - 需要提供 A100 SSH 密码 (见 Evo-RL/A100_login.txt)
#   - 脚本本身不存储任何密码
#   - 支持断点续传: 中断后重新运行即可继续
# =============================================================================

set -euo pipefail

# 从脚本位置向上推导 Evo-RL 根目录
SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
LOCAL_CODE_DIR="${SCRIPT_DIR}/.."   # Evo-RL/

A100_HOST="moganshan@180.184.148.169"
A100_PORT=10322
A100_CODE_DIR="/moganshan/afs_a/lai/Evo-RL"

echo ""
echo "============================================="
echo "  rsync: 代码同步到 A100"
echo "  本地:  ${LOCAL_CODE_DIR}"
echo "  远端:  ${A100_HOST}:${A100_CODE_DIR}/"
echo "  排除:  .git/ outputs/ pretrained/"
echo "  连接:  SSH 端口 ${A100_PORT}"
echo "============================================="
echo ""

rsync -avz --progress \
    --exclude='.git' \
    --exclude='outputs/' \
    --exclude='pretrained/' \
    "${LOCAL_CODE_DIR}/" \
    "${A100_HOST}:${A100_CODE_DIR}/" \
    -e "ssh -p ${A100_PORT}"

echo ""
echo "代码同步完成."
echo "后续步骤: 在 A100 上运行 setup_a100_env.sh 初始化 conda 环境"
```

- 排除 `.git`、`outputs/`、`pretrained/`，符合 D-10

```bash
#!/bin/bash
# =============================================================================
# setup_a100_env.sh — A100 conda 环境初始化引导脚本
#
# 这是一个本地运行的引导脚本，它将:
#   1. 生成一个远端执行脚本 (_a100_remote_env_setup.sh)
#   2. 通过 rsync 将该脚本复制到 A100
#   3. 打印用户需要在 A100 上手动执行的命令
#
# 为什么需要手动步骤:
#   su lai 需要交互式 TTY，无法在非交互式 SSH 中自动化执行。
#   用户必须 SSH 登录后手动切换账号。
#
# 用法:
#   bash scripts/setup_a100_env.sh
# =============================================================================

set -euo pipefail

A100_HOST="moganshan@180.184.148.169"
A100_PORT=10322
A100_CODE_DIR="/moganshan/afs_a/lai/Evo-RL"
A100_ENV_NAME="evo-rl"

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
REMOTE_SETUP_SCRIPT="${SCRIPT_DIR}/_a100_remote_env_setup.sh"

echo ""
echo "============================================="
echo "  A100 conda 环境初始化"
echo "  环境名: ${A100_ENV_NAME}"
echo "  安装路径: ${A100_CODE_DIR}"
echo "============================================="

# Step 1: 生成远端执行脚本
echo ""
echo "[1/3] 生成远端执行脚本..."
cat > "${REMOTE_SETUP_SCRIPT}" << 'REMOTE_SCRIPT'
#!/bin/bash
# _a100_remote_env_setup.sh — 在 A100 上运行 (su lai 之后执行)
# 由 setup_a100_env.sh 自动生成，请勿手动编辑

set -euo pipefail

A100_CODE_DIR="/moganshan/afs_a/lai/Evo-RL"
A100_ENV_NAME="evo-rl"

echo ""
echo "=== A100 conda 环境初始化 ==="
echo "环境名: ${A100_ENV_NAME}"
echo "代码路径: ${A100_CODE_DIR}"
echo ""

# 检查 conda 是否在 PATH
if ! command -v conda &> /dev/null; then
    echo "ERROR: conda 未找到，请先初始化 conda:"
    echo "  source ~/anaconda3/etc/profile.d/conda.sh"
    echo "  或者: source /opt/conda/etc/profile.d/conda.sh"
    exit 1
fi

# 激活 conda shell 支持
CONDA_BASE=$(conda info --base)
source "${CONDA_BASE}/etc/profile.d/conda.sh"

# D-21: 若环境已存在则跳过创建
if conda env list | grep -q "^${A100_ENV_NAME} "; then
    echo "conda 环境 '${A100_ENV_NAME}' 已存在，跳过创建."
else
    echo "创建 conda 环境 '${A100_ENV_NAME}' (Python 3.10)..."
    conda create -n "${A100_ENV_NAME}" python=3.10 -y
fi

conda activate "${A100_ENV_NAME}"

# 进入代码目录
if [ ! -d "${A100_CODE_DIR}" ]; then
    echo "ERROR: 代码目录不存在: ${A100_CODE_DIR}"
    echo "请先运行 rsync_to_a100.sh 同步代码"
    exit 1
fi

cd "${A100_CODE_DIR}"
echo "当前目录: $(pwd)"

# D-18: 安装基础依赖
echo ""
echo "安装基础依赖 (pip install -e .)..."
pip install -e .

# D-18 + Research finding: .[pi] 安装 PI0.5 所需的自定义 transformers 分支
echo ""
echo "安装 PI0.5 依赖 (pip install -e \".[pi]\")..."
echo "注意: 这将从 GitHub 拉取自定义 transformers 分支，需要 A100 能访问 github.com"
echo "      如果 GitHub 不可访问，请参考 pyproject.toml [project.optional-dependencies] pi 条目"
echo "      手动克隆 transformers fork 并从本地路径安装"
pip install -e ".[pi]"

# 验证安装
echo ""
echo "验证安装..."
python3 -c "
import torch
print(f'PyTorch: {torch.__version__}')
print(f'CUDA available: {torch.cuda.is_available()}')
if torch.cuda.is_available():
    n = torch.cuda.device_count()
    print(f'CUDA device count: {n}')
    for i in range(n):
        name = torch.cuda.get_device_name(i)
        mem = torch.cuda.get_device_properties(i).total_memory / 1e9
        print(f'  GPU {i}: {name} ({mem:.1f} GB)')
"

# 验证 lerobot 命令可用
python3 -c "import lerobot; print(f'lerobot: OK')"

echo ""
echo "============================================="
echo "  A100 环境初始化完成！"
echo ""
echo "  torchcodec 注意事项:"
echo "  如果后续运行 lerobot-value-infer 时出现 torchcodec 相关错误，"
echo "  执行以下命令修复:"
echo "    pip uninstall torchcodec -y"
echo "============================================="
REMOTE_SCRIPT

chmod +x "${REMOTE_SETUP_SCRIPT}"
echo "远端执行脚本已生成: ${REMOTE_SETUP_SCRIPT}"

# Step 2: 将脚本复制到 A100
echo ""
echo "[2/3] 将远端执行脚本复制到 A100..."
rsync -avz \
    "${REMOTE_SETUP_SCRIPT}" \
    "${A100_HOST}:${A100_CODE_DIR}/scripts/_a100_remote_env_setup.sh" \
    -e "ssh -p ${A100_PORT}"
echo "脚本已复制到 A100: ${A100_CODE_DIR}/scripts/_a100_remote_env_setup.sh"

# Step 3: 打印手动执行步骤
echo ""
echo "[3/3] 请按以下步骤在 A100 上完成环境初始化:"
echo ""
echo "============================================="
echo "  手动执行步骤"
echo "============================================="
echo ""
echo "1. SSH 登录 A100 (密码见 Evo-RL/A100_login.txt):"
echo "   ssh moganshan@180.184.148.169 -p ${A100_PORT}"
echo ""
echo "2. 切换到个人账号 (需要 lai 账号密码，见 A100_login.txt):"
echo "   su lai"
echo ""
echo "3. 运行环境初始化脚本:"
echo "   bash ${A100_CODE_DIR}/scripts/_a100_remote_env_setup.sh"
echo ""
echo "完成后 A100 上的 evo-rl conda 环境即可用于训练。"
echo ""
echo "注意: _a100_remote_env_setup.sh 已添加到 .gitignore 候选，"
echo "      不需要提交到版本控制（包含临时生成内容）"
echo "============================================="
```
