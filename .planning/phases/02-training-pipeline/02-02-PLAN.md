设计原则（来自 Research 决策）：
- A100 显存 80GB，无需像 4090 那样测试 batch=4 和 batch=8 两种配置
- 固定 batch=8，SMOKE_STEPS=10，三阶段各跑 10 步
- 输出清晰的通过/失败总结
- 路径配置与 train_pen_A100.sh 完全一致

```bash
#!/bin/bash
# =============================================================================
# Evo-RL Smoke Test — 4x A100 (80GB)
#
# 目的: 验证完整训练流水线在 A100 上能跑通 (每步只跑 10 步)
#       价值函数训练 → 价值推理 + ACP 标注 → PI0.5 策略训练
#
# A100 vs 4090 smoke test 的差异:
#   - A100 显存充裕 (80GB/卡)，无需测试多种 batch size
#   - 固定 batch=8，4 卡 DDP
#   - fold_cloth 任务使用全部 14 维 (双臂 + 3 相机)
#
# 使用方式:
#   conda activate evo-rl
#   bash scripts/smoke_test_A100.sh
#
# 后台运行:
#   nohup bash scripts/smoke_test_A100.sh > outputs/smoke_test_A100.log 2>&1 &
#   tail -f outputs/smoke_test_A100.log
# =============================================================================

set -euo pipefail

export TOKENIZERS_PARALLELISM=false

# A100 路径配置 (与 train_pen_A100.sh 保持一致)
WORK_DIR="/moganshan/afs_a/lai/Evo-RL"
cd "${WORK_DIR}"

DATA_DIR="/moganshan/afs_a/lai/data/fold_cloth"
PRETRAINED_DIR="/moganshan/afs_a/lai/pretrained"
OUTPUT_DIR="${WORK_DIR}/outputs/smoke_test_A100"
DATASET_REPO_ID="wzt/pen"

# =============================================================================
# GPU 配置
# =============================================================================
NUM_GPUS=4
LAUNCH="CUDA_VISIBLE_DEVICES=0,1,2,3 accelerate launch --multi_gpu --num_processes=${NUM_GPUS} --mixed_precision=bf16"

# =============================================================================
# Smoke test 参数 (最少步数，只验证能跑通)
# =============================================================================
SMOKE_STEPS=10

# =============================================================================
# 特征选择 — fold_cloth 任务使用全部 14 维 (D-24)
# 与 pen 任务不同 (pen 用 STATE_SLICE="0:7" 只取左臂)
# fold_cloth 为双臂任务，不排除相机，不截断维度
# =============================================================================
EXCLUDE_CAMERAS=""    # 保留全部 3 个相机
STATE_SLICE=""        # 使用全部 14 维状态
ACTION_SLICE=""       # 使用全部 14 维动作

mkdir -p "${OUTPUT_DIR}"

# =============================================================================
# 启动前检查
# =============================================================================
echo ""
echo "============================================="
echo "  Evo-RL Smoke Test (4x A100)"
echo "  代码:       ${WORK_DIR}"
echo "  数据:       ${DATA_DIR}"
echo "  预训练权重: ${PRETRAINED_DIR}"
echo "  输出:       ${OUTPUT_DIR}"
echo "  GPU:        ${NUM_GPUS} (CUDA 0,1,2,3)"
echo "  步数:       ${SMOKE_STEPS} steps/stage"
echo "  特征选择:   排除相机=${EXCLUDE_CAMERAS:-无}"
echo "              状态切片=${STATE_SLICE:-全部14维}"
echo "              动作切片=${ACTION_SLICE:-全部14维}"
echo "============================================="

# 检查 CUDA
python3 -c "
import torch
assert torch.cuda.is_available(), 'CUDA not available!'
n = torch.cuda.device_count()
print(f'CUDA OK: {n} GPU(s)')
for i in range(n):
    name = torch.cuda.get_device_name(i)
    mem = torch.cuda.get_device_properties(i).total_memory / 1e9
    print(f'  GPU {i}: {name} ({mem:.1f} GB)')
"

# 检查预训练权重
if [ ! -d "${PRETRAINED_DIR}/pi05_base" ]; then
    echo "ERROR: 预训练权重未找到: ${PRETRAINED_DIR}/pi05_base"
    echo "运行 rsync_weights_to_a100.sh 同步权重"
    exit 1
fi

# 检查数据集
if [ ! -f "${DATA_DIR}/meta/info.json" ]; then
    echo "ERROR: 数据集未找到: ${DATA_DIR}"
    echo "数据集需要单独上传到 A100 (Phase 2 暂未包含此步骤)"
    exit 1
fi

# =============================================================================
# Step 1/3: 价值函数训练 (10 步)
# =============================================================================
echo ""
echo "[Step 1/3] 价值函数训练 (pistar06, ${SMOKE_STEPS} 步, batch=${VALUE_TRAIN_BATCH:-8}/卡, ${NUM_GPUS} 卡)..."
echo "  实际 batch size: $((8 * NUM_GPUS))"
echo "---------------------------------------------"

rm -rf "${OUTPUT_DIR}/value_train"

eval ${LAUNCH} $(which lerobot-value-train) \
  --dataset.repo_id="${DATASET_REPO_ID}" \
  --dataset.root="${DATA_DIR}" \
  --value.type=pistar06 \
  --value.dtype=bfloat16 \
  --value.vision_repo_id="${PRETRAINED_DIR}/siglip-so400m-patch14-384" \
  --value.language_repo_id="${PRETRAINED_DIR}/gemma-3-270m" \
  --batch_size=8 \
  --steps="${SMOKE_STEPS}" \
  --save_freq="${SMOKE_STEPS}" \
  ${EXCLUDE_CAMERAS:+--train_exclude_cameras="${EXCLUDE_CAMERAS}"} \
  ${STATE_SLICE:+--train_state_slice="${STATE_SLICE}"} \
  --output_dir="${OUTPUT_DIR}/value_train" \
  --job_name=smoke_value_A100

echo "[Step 1/3] 价值函数训练完成."

# =============================================================================
# Step 2/3: 价值推理 + ACP 标注 (10 步)
#
# 注意: 如果此步骤因 torchcodec 错误失败，执行: pip uninstall torchcodec -y
# =============================================================================
echo ""
echo "[Step 2/3] 价值推理 + ACP 标注..."
echo "  注意: 如遇 torchcodec 错误，运行: pip uninstall torchcodec -y"
echo "---------------------------------------------"

rm -rf "${OUTPUT_DIR}/value_infer"

eval ${LAUNCH} $(which lerobot-value-infer) \
  --dataset.repo_id="${DATASET_REPO_ID}" \
  --dataset.root="${DATA_DIR}" \
  --inference.checkpoint_path="${OUTPUT_DIR}/value_train" \
  --runtime.batch_size=8 \
  --acp.enable=true \
  --acp.n_step=50 \
  --acp.positive_ratio=0.3 \
  --acp.value_field="complementary_info.value_smoke" \
  --acp.advantage_field="complementary_info.advantage_smoke" \
  --acp.indicator_field="complementary_info.acp_indicator_smoke" \
  --output_dir="${OUTPUT_DIR}/value_infer" \
  --job_name=smoke_infer_A100

echo "[Step 2/3] 价值推理完成."

# =============================================================================
# Step 3/3: PI0.5 策略训练 (10 步)
# =============================================================================
echo ""
echo "[Step 3/3] PI0.5 策略训练 (${SMOKE_STEPS} 步, full param, ${NUM_GPUS} 卡)..."
echo "  实际 batch size: $((8 * NUM_GPUS))"
echo "---------------------------------------------"

rm -rf "${OUTPUT_DIR}/pi05_train"

eval ${LAUNCH} $(which lerobot-train) \
  --dataset.repo_id="${DATASET_REPO_ID}" \
  --dataset.root="${DATA_DIR}" \
  --policy.type=pi05 \
  --policy.pretrained_path="${PRETRAINED_DIR}/pi05_base" \
  --policy.push_to_hub=false \
  --policy.dtype=bfloat16 \
  --policy.gradient_checkpointing=true \
  --policy.train_expert_only=false \
  --batch_size=8 \
  --steps="${SMOKE_STEPS}" \
  --num_workers=4 \
  --save_freq="${SMOKE_STEPS}" \
  --log_freq=5 \
  --acp.enable=true \
  --acp.indicator_field="complementary_info.acp_indicator_smoke" \
  --acp.indicator_dropout_prob=0.3 \
  ${EXCLUDE_CAMERAS:+--train_exclude_cameras="${EXCLUDE_CAMERAS}"} \
  ${STATE_SLICE:+--train_state_slice="${STATE_SLICE}"} \
  ${ACTION_SLICE:+--train_action_slice="${ACTION_SLICE}"} \
  --output_dir="${OUTPUT_DIR}/pi05_train" \
  --job_name=smoke_pi05_A100

echo "[Step 3/3] PI0.5 策略训练完成."

# =============================================================================
# 结果汇总
# =============================================================================
echo ""
echo "============================================="
echo "  Smoke Test 完成"
echo "  输出目录: ${OUTPUT_DIR}/"
echo "============================================="
echo ""
echo "结果:"
for d in "${OUTPUT_DIR}"/*/; do
    name=$(basename "$d")
    if [ -d "$d/checkpoints" ]; then
        echo "  ${name}: 通过"
    else
        echo "  ${name}: 失败或未完成"
    fi
done

echo ""
echo "下一步: 运行完整训练 bash scripts/train_pen_A100.sh"
```

- EXCLUDE_CAMERAS=""、STATE_SLICE=""、ACTION_SLICE=""（D-24，fold_cloth 全 14 维）
- value-infer 前有 torchcodec 警告注释（Pitfall 3）
- 每阶段 rm -rf 清除旧输出，确保干净运行
- smoke 专用 ACP 字段名（`_smoke` 后缀，避免污染 Round 1 的 `_r1` 字段）
